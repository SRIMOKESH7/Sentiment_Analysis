# Load data from CSV file with the appropriate encoding
data = pd.read_excel('pos_neg_eng_dataset.xlsx')

# Extract the text and sentiment labels from the CSV
texts = data['text'].tolist()
labels = data['sentiment'].tolist()

# Preprocess the text data
preprocessed_texts = [text.lower() for text in texts]

# Feature extraction using TF-IDF
vectorizer = TfidfVectorizer()
features = vectorizer.fit_transform(preprocessed_texts)

# Splitting the dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(features, labels, test_size=0.2, random_state=42)

# Training the Naive Bayes classifier
clf = MultinomialNB()
clf.fit(X_train, y_train)
